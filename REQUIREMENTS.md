  // 1. –ü—Ä–∏–º–µ–Ω–∏—Ç—å —Ñ–∏–ª—å—Ç—Ä –≥–ª–∞–≤—ã (–µ—Å–ª–∏ –≤—ã–±—Ä–∞–Ω)

    ? questions.filter(q => q.chapter === filter.chapterNumber)

  // 2. –ü–æ–ª—É—á–∏—Ç—å —Å—Ç–∞—Ç–∏—Å—Ç–∏–∫—É –¥–ª—è –≤–æ–ø—Ä–æ—Å–æ–≤

    const score = stats?.answerScore ?? 0; // null ‚Üí 0
    const weight = Math.pow(5 - score, 2);

    return { question: q, weight };
  });

  // 3. –í–∑–≤–µ—à–µ–Ω–Ω—ã–π —Å–ª—É—á–∞–π–Ω—ã–π –≤—ã–±–æ—Ä
  const totalWeight = questionsWithWeights.reduce((sum, q) => sum + q.weight, 0);

  if (totalWeight === 0) {
    // –í—Å–µ –≤–æ–ø—Ä–æ—Å—ã –∏–º–µ—é—Ç –æ—Ü–µ–Ω–∫—É 5 - –≤—ã–±—Ä–∞—Ç—å —Å–ª—É—á–∞–π–Ω–æ
    return filteredQuestions[Math.floor(Math.random() * filteredQuestions.length)];
  }

  let random = Math.random() * totalWeight;
  for (const item of questionsWithWeights) {
    random -= item.weight;
    if (random <= 0) {
      return item.question;
    }
  }

  return questionsWithWeights[0].question; // fallback
}
